                  using System;
                  using System.IO;
                  namespace BM.API.StaticFiles
                  {
                      public class FastPageGenerator
                      {
                          public static void Main()
                          {
                              //DeleteOldPages();
                              Generate("wwwroot");
                          }
                  
                          public static void Generate(string webRootPath)
                          {
                              DeleteOldPages(webRootPath);
                  
                              var includeDirectory = Path.Combine("StaticFiles", "include");
                              var contentDirectory = Path.Combine("StaticFiles", "content");
                  
                              var includePaths = Directory.GetFiles(includeDirectory, "*.shtml", SearchOption.AllDirectories);
                              var contentPaths = Directory.GetFiles(contentDirectory, "*.html", SearchOption.AllDirectories);
                              foreach (var content in contentPaths)
                              {
                                  var htmlString = File.ReadAllText(content);
                                  if (string.IsNullOrWhiteSpace(htmlString)) continue;
                                  var htmlName = Path.GetFileName(content);
                                  foreach (var path in includePaths)
                                  {
                                      var includeName = Path.GetFileName(path);
                  
                                      //var tag = $"";
                                      var tag = "";
                                      if (htmlString.Contains(tag))
                                      {
                                          var include = File.ReadAllText(path);
                                          htmlString = htmlString.Replace(tag, include);
                                      }
                                  }
                                  File.WriteAllText(Path.Combine(webRootPath, htmlName), htmlString);
                  
                              }
                          }
                  
                          private static void DeleteOldPages(string path)
                          {
                              var htmlFiles = Directory.GetFiles(path, "*.html", SearchOption.AllDirectories);
                              foreach (var file in htmlFiles)
                              {
                                  if (!Path.GetFileName(file).StartsWith("404"))
                                      File.Delete(file);
                              }
                          }
                      }
                  }
